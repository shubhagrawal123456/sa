			if(calculator.display.value.substring(0,1)=="-")

		function sin(val){
           if (val=="sin") {
              var x=document.getElementById("scn").value;
              x=x*Math.PI/180;
              document.getElementById("scn").value=Math.sin(x);
           }
       }
       function cos(val){
           if(val=="cos"){
           	var x=document.getElementById("scn").value;
           	x=x*Math.PI/180;
           	document.getElementById("scn").value=Math.cos(x);
           }
		}
		function tan(val){
			if (val=="tan"){
				var x=document.getElementById("scn").value;
				x=x*Math.PI/180;
				document.getElementById("scn").value=Math.tan(x);
			}
		}
